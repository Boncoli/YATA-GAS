# ✅ STEP 2：Method Descriptor抽出関数追加

（LlmServiceの外）

```js
function extractMethodDescriptor(title, abstractText){

 const systemPrompt = `
Extract ONLY the measurement or assay methodology.
Return JSON:
{
"principle":"",
"detection":"",
"sample":"",
"automation":"",
"instrumentation":""
}
`;

 const userPrompt = `
Title: ${title}
Abstract: ${abstractText}
`;

 const json = LlmService.analyzeKeywordSearch(
   systemPrompt,
   userPrompt,
   {temperature:0.0}
 );

 const parsed = cleanAndParseJSON(json);
 if(!parsed) return null;

 return `
Principle:${parsed.principle}
Detection:${parsed.detection}
Sample:${parsed.sample}
Automation:${parsed.automation}
Instrumentation:${parsed.instrumentation}
 `;
}
```

# ✅ STEP 3：Method Embedding生成関数

```js
function generateMethodVector(title, abstractText){

 const descriptor = extractMethodDescriptor(title, abstractText);
 if(!descriptor) return null;

 return LlmService.generateVector(descriptor);
}
```

```js

 rowArray[vecColIdx] = topicVector.join(',');
```
